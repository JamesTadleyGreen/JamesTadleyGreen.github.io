1 Chapter Exercises

1.1 Data.Char

    1. isUpper :: Char -> Bool
    2. toUpper :: Char -> Char
  2. filterUpper :: String -> String
filterUpper = filter isUpper
  3. capitalise :: String -> String
capitalise (x : xs) = toUpper x : xs
  4. shout :: String -> String
shout [] = []
shout (x : xs) = toUpper x : shout xs

  5. Note: This isn’t a complete function, fails on []

  6. capitaliseHead :: String -> Char
capitaliseHead = toUpper . head

2 Ciphers

  1. caeserChar :: Int -> Char -> Char
caeserChar rot c = chr $ 97 + mod (ord c + rot - 97) 26

caeser :: Int -> String -> String
caeser rot s = map (caeserChar rot) s

unCaeser :: Int -> String -> String
unCaeser rot = caeser (-rot)

3 Writing our own standard functions

  1. myOr :: [Bool] -> Bool
myOr [] = False
myOr (x : xs) = x || myOr xs
  2. myAny :: (a -> Bool) -> [a] -> Bool
myAny f xs = myOr $ map f xs
  3. myElem :: (Eq a) => a -> [a] -> Bool
myElem a xs = myAny (== a) xs
  4. myReverse :: [a] -> [a]
myReverse [] = []
myReverse (x : xs) = myReverse xs ++ [x]
  5. squish :: [[a]] -> [a]
squish [] = []
squish (x : xs) = x ++ squish xs
  6. squishMap :: (a -> [b]) -> [a] -> [b]
squishMap f [] = []
squishMap f (x : xs) = f x ++ squishMap f xs
  7. squishAgain :: [[a]] -> [a]
squishAgain = squishMap id
  8. myMaximumBy :: (a -> a -> Ordering) -> [a] -> a
myMaximumBy f (x : xs) = foldl (comp f) x xs
  where
    comp f a b
      | f a b == LT = b
      | otherwise = a
  9. myMinimumBy :: (a -> a -> Ordering) -> [a] -> a
myMinimumBy f = myMaximumBy (invert f)
  where
    invert comp a b
      | comp a b == LT = GT
      | otherwise = LT
  10. myMaximum :: (Ord a) => [a] -> a
myMaximum = myMaximumBy compare
  11. myMinimum :: (Ord a) => [a] -> a
myMinimum = myMinimumBy compare
